## NPM
```sh
$ npm install bloomer --save
```

## Yarn
```sh
$ yarn add bloomer
```

## Usage
```javascript
// you should have bulma in your project
import * as React from 'react';
import * as ReactDOM from 'react-dom';
import { Container, Box  } from 'bloomer';

ReactDOM.render(
    <Container>
        <Box>Hello World!</Box>
    </Container>,
    document.getElementById('root')
)
```
